Examples

Sharpen filter

var filter = new fabric.Image.filters.Convolute({
  matrix: [ 0, -1,  0,
           -1,  5, -1,
            0, -1,  0 ]
});
object.filters.push(filter);
object.applyFilters(canvas.renderAll.bind(canvas));

Blur filter

var filter = new fabric.Image.filters.Convolute({
  matrix: [ 1/9, 1/9, 1/9,
            1/9, 1/9, 1/9,
            1/9, 1/9, 1/9 ]
});
object.filters.push(filter);
object.applyFilters(canvas.renderAll.bind(canvas));

Emboss filter

var filter = new fabric.Image.filters.Convolute({
  matrix: [ 1,   1,  1,
            1, 0.7, -1,
           -1,  -1, -1 ]
});
object.filters.push(filter);
object.applyFilters(canvas.renderAll.bind(canvas));

Emboss filter with opaqueness

var filter = new fabric.Image.filters.Convolute({
  opaque: true,
  matrix: [ 1,   1,  1,
            1, 0.7, -1,
           -1,  -1, -1 ]
});
object.filters.push(filter);
object.applyFilters(canvas.renderAll.bind(canvas));
